jQuery(function($){
	var widget = $("#widget"),
	throbber = $('<img src="throbber.gif" alt="Loading..." width="54" height="55" />').appendTo(widget);
	wowHeroes.getGuild("eu", "outland", "Mysth", function(characters){
		throbber.remove();

		if(characters){
			widget.append("<table><thead><tr><th>C</th><th>Name</th><th>Score</th></tr></thead><tbody></tbody></table>");
			var tbody = widget.find('tbody');
			$.each(characters, function(i, c){
				tbody.append("<tr><td><img src=\"classicons/"+c.classId+".gif\" alt=\""+c.classId+"\" width=\"18\" height=\"18\"/></td><td>"+
					"<a class=\"class_"+c.classId+"\" href=\"http://wow-heroes.com/index.php?zone=eu&server=Outland&name="+escape(c.name)+"\">"+c.name+"</a>"+
					"</td><td>"+c.score+"</td></tr>");
			});
			widget.find('table').tablesorter();
		}
	});
});
